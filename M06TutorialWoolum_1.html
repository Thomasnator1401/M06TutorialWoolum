// Import the MongoDB module const mongodb = require('mongodb'); // Create a MongoClient object const MongoClient = mongodb.MongoClient; // Define the connection URL and database name const url = 'mongodb://localhost:27017'; const dbName = 'mydb'; // Connect to the server MongoClient.connect(url, function(err, client) { if (err) { console.error(err); } else { console.log('Connected successfully to server'); // Get the database object const db = client.db(dbName); // Create a collection named users const users = db.collection('users'); // Insert some documents into the collection users.insertMany([ {name: 'Alice', age: 25}, {name: 'Bob', age: 30}, {name: 'Charlie', age: 35} ], function(err, result) { if (err) { console.error(err); } else { console.log('Inserted ' + result.insertedCount + ' documents into the collection'); } // Close the connection client.close(); }); } });
